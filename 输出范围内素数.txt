<script>
	function isPrime(n){
		if (n == 1) {
			return false;
			}else{
			var flag = true;
			for (var i =2; i<n; i++){
				if(n%i==0){
					flag = false;
					break;
					}
				}
			return flag;
			}
		}
	function showPrime( start, end ){
		for (var i=start; i<=end; i++){
			if (isPrime(i)){
				document.write(i + '<br>');
				}
			}
		}
	showPrime(10,100);
</script>
